Then for `handle_query`:
```rust
async fn handle_query(State(graph): State<SharedGraph>, body: String) -> impl IntoResponse {
    let cancel = Arc::new(std::sync::atomic::AtomicBool::new(false));
    let _guard = CancelGuard(cancel.clone());

    let mut g = graph.lock_owned().await;
    g.cancel_flag = cancel;

    let res = tokio::task::spawn_blocking(move || {
        let result = g.execute(&body);
        g.cancel_flag = Arc::new(std::sync::atomic::AtomicBool::new(false));
        result
    }).await;

    match res {
        Ok(Ok(result)) => (StatusCode::OK, result).into_response(),
        Ok(Err(e)) => (StatusCode::BAD_REQUEST, format!("Error: {}", e)).into_response(),
        Err(_) => (StatusCode::INTERNAL_SERVER_ERROR, "Server error".to_string()).into_response(),
    }
}
```
